## Lock
```js
// 无权访问的组件类型
export enum LockType {
    'PAGE',//展示403页面
    'BUTTON',//disabled
    'MENU'//不展示
}
interface LockProps {
    children: React.ReactElement;
    type: LockType;
    permissionCheck?:Function,//校验函数
    ERROR_403?:ReactNode//403元素
}
```
